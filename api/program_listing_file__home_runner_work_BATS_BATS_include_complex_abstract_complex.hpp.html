// indicates dimension and index of a cell
struct cell_ind {
public:
    size_t dim;
    size_t ind;

    cell_ind() {}
    cell_ind(size_t dim, size_t ind) : dim(dim), ind(ind) {}

    std::string str() {
        std::ostringstream oss;
        oss << '(' << dim << ',' << ind << ')';
        return oss.str();
    }
};
